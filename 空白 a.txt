        // --- CORE UTILS ---
        function calculate() {
            const input = document.getElementById('calc-input').value;
            try {
                // Safe calculation using new Function
                const result = new Function(`return ${input}`)();
                document.getElementById('calc-jpy').innerText = `¥ ${Math.round(result).toLocaleString()}`;
                document.getElementById('calc-twd').innerText = `≈ NT$ ${Math.round(result * JPY_RATE).toLocaleString()}`;
            } catch (e) { alert("Invalid Calculation"); }
        }

        function updateRate(val) {
            JPY_RATE = parseFloat(val);
            localStorage.setItem('jpy_rate', JPY_RATE);
            renderWallet(document.getElementById('app-content'));
        }

        let tempImageBase64 = null;
        function handleImage(input) {
            const file = input.files[0];
            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    // Resize logic: max 300px
                    const scale = 300 / Math.max(img.width, img.height);
                    canvas.width = img.width * scale;
                    canvas.height = img.height * scale;
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    tempImageBase64 = canvas.toDataURL('image/jpeg', 0.7); // Compress
                    document.getElementById('temp-img').src = tempImageBase64;
                    document.getElementById('image-preview').classList.remove('hidden');
                };
                img.src = e.target.result;
            };
            reader.readAsDataURL(file);
        }

        function saveExpense() {
            const name = document.getElementById('exp-name').value;
            const jpy = document.getElementById('exp-amount').value;
            if (!name || !jpy) return;
            
            const expenses = JSON.parse(localStorage.getItem('expenses') || '[]');
            expenses.push({
                name, jpy: parseFloat(jpy), img: tempImageBase64,
                date: new Date().toLocaleDateString('zh-TW', {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'})
            });
            localStorage.setItem('expenses', JSON.stringify(expenses));
            tempImageBase64 = null;
            renderWallet(document.getElementById('app-content'));
        }

        function deleteExpense(i) {
            const expenses = JSON.parse(localStorage.getItem('expenses') || '[]');
            expenses.splice(i, 1);
            localStorage.setItem('expenses', JSON.stringify(expenses));
            renderWallet(document.getElementById('app-content'));
        }

        function saveMemo(val) {
            localStorage.setItem('user_memo', val);
            // Throttle rendering for link buttons
            clearTimeout(window.memoTimer);
            window.memoTimer = setTimeout(() => renderLists(document.getElementById('app-content')), 1000);
        }

        // Init
        window.onload = () => renderPage('plan');
    </script>
