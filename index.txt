        function calculate() {
            const input = document.getElementById('calc-input').value;
            try {
                const res = Function('"use strict";return (' + input + ')')();
                document.getElementById('calc-jpy').innerText = `¥ ${Math.round(res)}`;
                document.getElementById('calc-twd').innerText = `≈ NT$ ${Math.round(res * JPY_RATE)}`;
            } catch (e) { alert("計算格式錯誤"); }
        }

        function processImage(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    const img = new Image();
                    img.src = e.target.result;
                    img.onload = () => {
                        const canvas = document.createElement('canvas');
                        const MAX_W = 300;
                        const scale = MAX_W / img.width;
                        canvas.width = MAX_W;
                        canvas.height = img.height * scale;
                        canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height);
                        document.getElementById('exp-img-data').value = canvas.toDataURL('image/jpeg', 0.6);
                    };
                };
                reader.readAsDataURL(input.files[0]);
            }
        }

        function saveExpense() {
            const name = document.getElementById('exp-name').value;
            const jpy = document.getElementById('exp-amount').value;
            const img = document.getElementById('exp-img-data').value;
            if (!name || !jpy) return alert("請填入名稱與金額");
            const history = JSON.parse(localStorage.getItem('expenses') || '[]');
            history.push({ name, jpy, img, date: new Date().toLocaleString('zh-TW', {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'}) });
            localStorage.setItem('expenses', JSON.stringify(history));
            renderWallet(document.getElementById('app-content'));
        }

        function deleteExpense(i) {
            const history = JSON.parse(localStorage.getItem('expenses') || '[]');
            history.splice(i, 1);
            localStorage.setItem('expenses', JSON.stringify(history));
            renderWallet(document.getElementById('app-content'));
        }

        function saveNotes() {
            const text = document.getElementById('note-area').value;
            localStorage.setItem('user_notes', text);
            updateLinkButtons();
        }

        function updateLinkButtons() {
            const container = document.getElementById('links-preview');
            const text = document.getElementById('note-area')?.value || "";
            const urls = text.match(/(https?:\/\/[^\s]+)/g);
            if (container) {
                container.innerHTML = '';
                if (urls) {
                    urls.forEach(url => {
                        container.innerHTML += `<a href="${url}" target="_blank" class="text-[10px] bg-gold text-white px-3 py-1 rounded-full font-bold shadow-sm">OPEN LINK <i class="fa-solid fa-arrow-up-right-from-square ml-1"></i></a>`;
                    });
                }
            }
        }
